### 3. 데이터 확인 쿼리

```sql
-- 1. 전체 통계
SELECT
    'brands' as 테이블, COUNT(*) as 개수 FROM brands
UNION ALL
SELECT 'products', COUNT(*) FROM products
UNION ALL
SELECT 'personas', COUNT(*) FROM personas
UNION ALL
SELECT 'product_personas', COUNT(*) FROM product_personas;

-- 2. 브랜드별 상품 수
SELECT
    b.name AS 브랜드명,
    COUNT(p.id) AS 상품수
FROM brands b
LEFT JOIN products p ON b.id = p.brand_id
GROUP BY b.name
ORDER BY 상품수 DESC
LIMIT 20;

-- 3. 가격대별 상품 분포
SELECT
    CASE
        WHEN sale_price < 10000 THEN '1만원 미만'
        WHEN sale_price < 30000 THEN '1~3만원'
        WHEN sale_price < 50000 THEN '3~5만원'
        WHEN sale_price < 100000 THEN '5~10만원'
        ELSE '10만원 이상'
    END AS 가격대,
    COUNT(*) AS 상품수,
    ROUND(AVG(sale_price), 0) AS 평균가격
FROM products
WHERE sale_price IS NOT NULL
GROUP BY 가격대
ORDER BY MIN(sale_price);

-- 4. 평점 높은 상품 (리뷰 10개 이상)
SELECT
    b.name AS 브랜드,
    p.product_name AS 상품명,
    p.rating AS 별점,
    p.review_count AS 리뷰수,
    p.sale_price AS 판매가
FROM products p
JOIN brands b ON p.brand_id = b.id
WHERE p.review_count >= 10
ORDER BY p.rating DESC, p.review_count DESC
LIMIT 20;

-- 5. 페르소나 태그 통계
SELECT
    unnest(skin_types) AS 피부타입,
    COUNT(*) AS 상품수
FROM products
WHERE array_length(skin_types, 1) > 0
GROUP BY 피부타입
ORDER BY 상품수 DESC;

SELECT
    unnest(concern_keywords) AS 고민키워드,
    COUNT(*) AS 상품수
FROM products
WHERE array_length(concern_keywords, 1) > 0
GROUP BY 고민키워드
ORDER BY 상품수 DESC
LIMIT 20;

-- 6. 할인율이 높은 상품
SELECT
    b.name AS 브랜드,
    p.product_name AS 상품명,
    p.original_price AS 원가,
    p.discount_rate AS 할인율,
    p.sale_price AS 판매가,
    (p.original_price - p.sale_price) AS 할인금액
FROM products p
JOIN brands b ON p.brand_id = b.id
WHERE p.discount_rate > 0
ORDER BY p.discount_rate DESC
LIMIT 20;
```
